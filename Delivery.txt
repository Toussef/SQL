CREATE TABLE  "DELIVERY" 
   (	"ID" NUMBER(8,0), 
	"VEHICLE" VARCHAR2(100) NOT NULL ENABLE, 
	"HOUR" NUMBER(*,0) NOT NULL ENABLE, 
	"DAY" NUMBER(2,0) NOT NULL ENABLE, 
	"MONTH" NUMBER(2,0) NOT NULL ENABLE, 
	"YEAR" NUMBER(4,0) NOT NULL ENABLE, 
	"STATUS" VARCHAR2(5) NOT NULL ENABLE, 
	 PRIMARY KEY ("ID") ENABLE
   ) ;
